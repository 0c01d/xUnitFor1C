&НаКлиенте
Функция ПолучитьСписокТестов(ЮнитТестирование) Экспорт
	
	ЮТест = ЮнитТестирование;
	
	ВсеТесты = Новый Массив;

	ПараметрыТеста = Новый Структура("ПредставлениеТеста,ИмяТеста,Параметр", "Тест открытия инструмента ""Генератор данных""", "ТестДолжен_ОткрытьЗакрытьИнструмент", "xddDataFixtureGen.epf");
	ВсеТесты.Добавить(ПараметрыТеста);

	ПараметрыТеста = Новый Структура("ПредставлениеТеста,ИмяТеста,Параметр", "Тест открытия инструмента ""Показать ГУИД""", "ТестДолжен_ОткрытьЗакрытьИнструмент", "xddGuidShow.epf");
	ВсеТесты.Добавить(ПараметрыТеста);

	Возврат ВсеТесты;
	
КонецФункции
//}

//{ блок юнит-тестов - сами тесты

&НаКлиенте
Процедура ТестДолжен_ОткрытьЗакрытьИнструмент(ИмяИнструмента) Экспорт
	ОткрытаяФорма = ЮТест.ОткрытьИнструмент(ИмяИнструмента);
	ЮТест.ПроверитьТип(ОткрытаяФорма, "УправляемаяФорма", "Не удалось получить управляемую форму инструмента <"+ИмяИнструмента+">");
	ЮТест.Проверить(ОткрытаяФорма.Открыта(), "Форма инструмента должна быть открыта. Инструмент <"+ИмяИнструмента+">");
	ОткрытаяФорма.Закрыть();
	ОткрытаяФорма = Неопределено;
КонецПроцедуры
